!Функция для обработки строки с переменными, в частности, нужна для обработки переменных переданных через JS.
!Проблема в том, что JS не умеет напрямую работать с переменными QSP, он не умеет их ни читать не назначать.
!Поэтому - единственный способ передачи данных из JS в QSP происходит через <a> тэг, т.к. A-тэг это по HTML элемент (один из немногих, с которыми умеет работать QSP).
!В нашей ситуации, мы через JS можем создать новый a-элемент и впилить его в DOM, а QSP сможет его прочитать.
!А уже сами переменные - мы назначаем через href="exec:"
!Но, проблема в том, что кол-во переменных, которые можно передать в exec - ограничено (Точное кол-во не помню.)
!Именно для таких ситуаций нам и нужна эта функция.

!Эта функция принимает 2 аргумента:
!1. Комманда, которую нужно выполнить, например PARSE_VAR_STR для обработки строки с переменными.
!2. Строку с переменными, которые нужно обработать.

!Пример использования:
!Создаём A-тэг, например такой: <a href='exec:gs(''js_parser'',''parse_var_str'',''СЮДА_ВСТАВЛЯЕМ_СТРОКУ_С_ПЕРЕМЕННЫМИ'') '>click me!</a>
!Строка с переменными должна выглядеть примерно так -> somevar+=1,$othervar=`sometext`,anothervar[`someindex`]=35,
!ВАЖНО: Строки и текстовые индексы должны быть обёрнуты в такие -> `` кавычки, они будут заменены обычными в цикле.
!Не буду объяснять зачем это нужно, кто знает тот знает, кто не знает - прочитайте документацию по работе со строками.





$ARGS[0]=UCASE($ARGS[0])


if $ARGS[0] = 'PARSE_VAR_STR':
  if $ARGS[1] ! '':
    !заменяем кавычки(Если нужно).
    $ARGS['VAR_LIST']=$REPLACE($ARGS[1], '`', '"')
    $ARGS['VAR_LIST']=$TRIM($ARGS['VAR_LIST'])

    !Проверяем на наличие запятой в конце, если нет - добавляем.
    $ARGS['COMMA_CHECK']=$MID($ARGS['VAR_LIST'], LEN($ARGS['VAR_LIST']), 1)
    if $ARGS['COMMA_CHECK'] ! ',':$ARGS['VAR_LIST']+= ','

    :string_parser

    ARGS['MARK'] = 0

    !Ищем первую запятую и записываем её индекс.
    ARGS['MARK'] = INSTR($ARGS['VAR_LIST'], ",")

    !Если нашли запятую - запускаем процесс.
    if ARGS['MARK'] ! 0:

    !Записываем значение до запятой как результат.
    $ARGS['VAR_NAME'] = $MID($ARGS['VAR_LIST'], 1, ARGS['MARK'])

    !Удаляем значение до запятой из оригинала.
    $ARGS['VAR_LIST'] = $MID($ARGS['VAR_LIST'], ARGS['MARK'] + 1)

    !Удаляем запятую из результата.
    $ARGS['VAR_NAME'] = $TRIM($REPLACE($ARGS['VAR_NAME'], ',', ''))

    dynamic('<<$ARGS["VAR_NAME"]>>')

    jump 'string_parser'

    end
  end
end
