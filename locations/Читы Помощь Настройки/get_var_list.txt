$ARGS[0]=$UCASE($ARGS[0])
!Список возможных переменных секции "BODY".
!Как это работает:
!Мы записываем всю актуальную информацию о параметре, его лимитах и степпинге в одну строку, из которой, мы в последствии создаём отдельный массив.

!Зачем это нужно? Можно-же было просто записать всё вручную!
!Всё верно, это можно было сделать и вручную, для каждой переменной - создать свою собственную кнопку в списке, добавив пачку if-else, и прописав лимиты.
!Но такая штука будет огромна в размерах, и трудно-читаемая, и если, вдруг, нам нужно будет добавить новый пункт - нам придётся копи-пастить целый блок кода.

!Здесь-же, мы просто создаём один генератор, который автоматически генерирует все кнопки, с уже правильно обозначенными if-else.
!Если нам нужно будет добавить ещё один параметр в список - мы просто можем добавить его в список ниже как $ARGS[] = "NEW,НОВЫЙ ПАРАМЕТР,100,10," и он автоматически будет отрисован.

!Как работают эти строки, если например взять $ARGS[] = "face,Феминность лица,100,10,"
!1. - "face" внутригровое имя переменной, т.е. в данном случае, массив face[] отвечает за параметр "феминность/мускулистость лица".
!2. - "Феминность лица" - название параметра, для интерпретатора.
!3. - "100" - лимит параметра, т.е. условно-максимальное значение параметра, во многих случаях ничто нам не мешает выйти за его пределы, но это не имеет смысла.
!4. - "10" - степпинг параметра, этот пункт отвечает за то, насколько мы будет увеличивать/уменьшать параметр за "шаг" -
!Т.к. разные параметры имеют разное максимальное значение - например FAT[] спокойно может быть выше 1000 или 2000, а размер глаз может быть от 0 до 3, т.е. в первом случае -
!Имеет смысл увеличивать параметр на 100+ за шаг а во втором только на 1.

!Далее, мы парсим эту строку, запятые в строке - являются своего-рода "разделителями", говорящими нам откуда до куда резать.
!После того, как мы получаем первый параметр, мы инициализируем новую переменную, если представить что $ARGS[0] = "BODY" и строка которую мы парсим $ARGS[] = "face,Феминность лица,100,10,"
!То мы создаём новую переменную $BODY_face[], и в нулевой индекс идёт значение "face", в $BODY_face[1] = "Феминность лица", в $BODY_face[2] = "100", в $BODY_face[3] = "10".
!И записываем "face" в $RESULT, т.е. $RESULT += "face,".

!Далее, мы можем использовать массив $BODY_face[] как нам угодно, например, зная какие массивы были созданы (нам $RESULT возвращает список), мы можем прогнать эти массивы в цикле и на основе этого отрисовать наши кнопки.

!Это предохранитель, На случай, если по ошибке будет передано больше одного аргумента.
if ARRSIZE('ARGS') = 1:
	!* -------------------------------------------------------------------------- *!
	!*                                 СЕКЦИЯ BODY                                *!
	!* -------------------------------------------------------------------------- *!
	if $ARGS[0] = "BODY":
	  $ARGS[] = "face,Феминность лица,100,10,"&!FACE - феминность/мускулистость лица, от 0 до 100, где 0 - максимальная мускулистость, 100 - максимальная феминность. (Степпинг 20 пунктов, 0-20-40-60-80+)
	  $ARGS[] = "skin,Цвет кожи,6,1,"&!SKIN - цвет кожи, возможные значения от 0 до 6, где 0 - бледная, 1 - светлая, 2 - загорелая, 3 - смуглая, 4 - шоколадная, 5 - темная, 6 - черная.
	  $ARGS[] = "hcol,Цвет волос,6,1,"&!HCOL - цвет волос, возможные значения от 0 до 6, где 0 - черные, 1 - русые, 2 - рыжие, 3 - светлые, 4 - синие, 5 - красные, 6 - белые.
	  $ARGS[] = "gla,Размер глаз,3,1,"&!GLA - размер глаз, возможные значения от 0 до 3, где 0 - узкие, 1 - маленькие, 2 - раскосые, 3 - большие, (100 - слепые.)
	  $ARGS[] = "glacol,Цвет глаз,6,1,"&!GLACOL - цвет глаз, возможные значения от 0 до 6, где 0 - черные, 1 - карие, 2 - серые, 3 - зеленые, 4 - голубые, 5 - красные, 6 - желтые.
	  $ARGS[] = "nose,Форма носа,5,1,"&!NOSE - форма носа, 0 - обыкновенный, 1 - маленький вздернутый носик, 2 - утонченный, аристократический нос, 3 - широкий, 4 - горбатый, 5 - длинный, крючковатый нос.
	  $ARGS[] = "scull,Форма скул,1,1,"&!SCULL -  форма скул, возможные значения от 0 до 1+, где 0 - обыкновенные скулы, 1+ - широкие скулы.
	  $ARGS[] = "cheeks,Форма щёк,1,1,"&!CHEEKS - форма щек, возможные значения от 0 до 1+, где 0 - обыкновенные щеки, 1+ - пухлые щеки.
	  $ARGS[] = "beard,Борода,100,10,"&!BEARD - наличие бороды, возможные значения от 0 до 100, где 0 - гладко выбрит, 100 - длинная борода. (Степпинг не ровномерный, 0-25-75+)
	  $ARGS[] = "mustage,Усы,1,1,"&!MUSTAGE - наличие усов, возможные значения от 0 до 1+, где 0 - нет усов, 1+ - есть усы.
	  $ARGS[] = "dick,Длинна члена,50,1,"&!DICK - длина члена, от 1 до ???, каждый пункт добавляет 1 см. (Не нашел хард-ресета (если значение больше чем ???) и/или лимитов, так что возможно их и нет.)
	  $ARGS[] = "dickd,Диаметр члена,100,10,"&!DICKD - диаметр члена, возможные значения от 1 до ???, каждые 10 пунктов добавляют 1 сантиметр в диаметре. (не нашел ресетов или лимитов.)
	  $ARGS[] = "ball,Размер яиц,100,25,"&!BALL - размер яиц, возможные значения от 1 до ???, где 0 - нет, 1-24 -  яйчишки, 25-49 - яйца, 50-74 - яйчища, 75+ - яйчуища. (лимитов и ресетов не нашел.)
	  $ARGS[] = "vag,Наличие вагины,1,1,"&!VAG - наличие вагины, возможные значения от 0 до 1+, где 0 - нет, 1+ - есть.
	  $ARGS[] = "anus,Состояние ануса,40,1,"&!ANUS - состояние ануса, возможные значения от 0 до 40+, где 0 - девственный анус, 40+ - разъебанная дыра в жопе. (Степпинг 10 пунктов, 0-1-10-20-30-40+)
	  $ARGS[] = "nip,Размер сосков,50,10,"& !NIP - размер сосков, значения от 1 до 50+, где 0 - крохотные соски, 50+ - огромные соски. (После 20, каждые 10 пунктов добавляют 1 сантиметр в диаметре.)
	  $ARGS[] = "tits,Размер груди,25,1,"&!TITS - размер груди, значения от 0 до 25+, 0 - плоская, 25+ - огромная грудь. (Шаги 0-10-14-17-19-21-23-25, где после 10, каждый шаг добавляет +1 размер груди.)
	  $ARGS[] = "hlong,Длинна волос,100,10,"&!HLONG - длина волос, значения от 0 до 75+, где 0 - лысая голова, 75+ - длинные волосы до поясницы. (Степпинг 25 пунктов, 0-25-50-75+)
	  $ARGS[] = "rost,Рост,250,5,"&!ROST - отвечает за рост, каждый пункт - 1 см.
	  $ARGS[] = "fat,Жир,5000,100,"&!FAT - вес, в сочитании с MUS определяет общую массу тела и тип тела. В самом низу выкладки по сочитаниям MUS + FAT, либо можно проверить это в локации body.
	  $ARGS[] = "mus,Мускулистость,5000,100,"&!MUS - Отвечает за мышечную массу персонажа, в сочитании с FAT определяет общую массу тела и тип тела..
	  $ARGS[] = "tit_hair,Волосы на груди,1,1,"& !TIT_HAIR - Отвечает за наличие волос на груди, возможные значения от 0 до ???, где 0 - нет, 1+ - на груди растут волосы.
	  $ARGS[] = "leg_hair,Волосы на ногах,1,1,"&!LEG_HAIR - Отвечает за наличе волос на ногах, возможные значения от 0 до 1+, где 0 - гладкие ноги, 1+ - ноги волосатые.
	end
	!* -------------------------------------------------------------------------- *!
	!*                              СЕКЦИЯ BODY - END                             *!
	!* -------------------------------------------------------------------------- *!
	!Если хотя-бы один параметр был назначен - запускаем цикл.
	if $ARGS[1] ! '' :
		i = ARRSIZE('ARGS') - 1
		:main_loop
		TEMP['MARK'] = 0&TEMP['VAR_INIT'] = 0&$TEMP['VAR_NAME'] = ''&!Временные переменные, мы не можем использовать для этого $ARGS[] т.к. он используется для строк выше.
		:splitter_loop
		TEMP['MARK'] = INSTR($ARGS[i], ",")&!Ищем первую запятую и записываем её индекс.!Если нашли запятую - запускаем процесс.
		if TEMP['MARK'] ! 0:
			$TEMP['VALUE'] = $MID($ARGS[i], 1, TEMP['MARK'])&!Записываем значение до запятой как результат.
			$ARGS[i] = $MID($ARGS[i], TEMP['MARK'] + 1)&!Удаляем значение до запятой из оригинала.
			$TEMP['VALUE'] = $REPLACE($TEMP['VALUE'], ',', '')&!Удаляем запятую из результата.
			!Если это первая итерация - инициализируем переменную и записываем <<$TEMP['VALUE']>> в нее.
			!так-же добавляем <<$TEMP['VALUE']>> в список $RESULT
			if $TEMP['VAR_NAME']='':
				$TEMP['VAR_NAME'] = $TEMP['VALUE']
				dynamic "$<<$ARGS[0]>>_<<$TEMP['VAR_NAME']>>[0] = '<<$TEMP['VALUE']>>' & $RESULT += '<<$TEMP['VALUE']>>,'"
				!Если это не первая итерация - добавляем <<$TEMP['VALUE']>> в ново-созданную переменную под индексами 1-2-3
			else
				dynamic "$<<$ARGS[0]>>_<<$TEMP['VAR_NAME']>>[] = '<<$TEMP['VALUE']>>'"
			end
			jump "splitter_loop"
		end
		if i>1:i-=1 & jump'main_loop'
		killvar 'TEMP'
	end
end
